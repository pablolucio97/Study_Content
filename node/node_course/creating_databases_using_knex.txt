1) Create the directory for your database migrations

2)Setup your fileknesfile.js as folow:

  development: {
    client: 'sqlite3',
    connection: {
      filename: './src/database/db.sqlite'
    },
    migrations: {
      directory: './src/database/migrations'
    },
    useNullAsDefault: true,
  },


3) Run the code: npx knex migrate:make create_nameOfTable

4) Create your table, for example:

//CREATING A NEW TABLE WITH KNEX
exports.up = function(knex) {
 return knex.schema.createTable('ongs', function(table){
      table.string('id').primary();

      table.string('email').notNullable();
      table.string('whatsapp').notNullable();
      table.string('city').notNullable();
      table.string('uf', 2).notNullable();
  })
};

exports.down = function(knex) {
 return  knex.schema.dropTable('ongs')
};

5)Run the below code to generate your table file:

npx knex migrate:latest
