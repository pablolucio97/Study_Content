import React, {useState, useEffect} from 'react'
import { View, StyleSheet } from 'react-native'
import { DefaultTheme, Provider as PaperProvider, FAB} from 'react-native-paper';
import {Button, Title, Subheading, Caption, TextInput, Dvidider} from 'react-native-paper'
import Icon from 'react-native-vector-icons/Feather'


const theme = {
  ...DefaultTheme,
  roundness: 2,
  colors:{
    ...DefaultTheme.colors,
    primary: '#52ff',
    acent: '#77ff'
  }

}


const App = () => {

  const [seconds, setSeconds] = useState(new Date())
  const [minutes, setMinutes] = useState(new Date())
  const [hours, setHours] = useState(new Date())


  const myClock = () => {
  setSeconds(seconds.getSeconds())
  setMinutes(minutes.getMinutes())
  setHours(hours.getHours())
  }


  useEffect(() => {
    const interval = setInterval(() => {
      myClock
    }, 1000)
    return () => {clearInterval(interval)}
  })


  


  return (
    <PaperProvider theme={theme}>
    <View style={styles.mainContainer}>
      <Title>Clock</Title>
  <Subheading>{`${hours}:${minutes}:${seconds}`}</Subheading>
  <Title style={{padding: 30, fontSize: 35}}></Title>
      <Icon
      name='clock'
      size={65}
      color= '#77bb'
      />
    </View>
    <View style={styles.adjacentContainer}>
    </View>
    </PaperProvider>
  )
}

const styles = StyleSheet.create({
  mainContainer: {
    flex: .2,
    flexDirection: 'column',
    padding: 10,
    alignItems: 'center',
  }
})


export default App
