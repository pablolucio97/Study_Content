import React, { Component } from 'react'
import { Text, View, Button, StatusBar, Switch, StyleSheet } from 'react-native'

export default class App extends Component {
  constructor(props){
    super(props);
    this.state = {
      time: 10,
      switcher: false,
      switcher2: false,

    }
  }

  startCount = () => {  
    let time = this.state.time
    this.setState({time: this.state.time -1})
    if(time <= 0 && this.state.switcher === true ){
      alert('Time finish')
      this.setState({time: this.state.time = 0})
    }else if (time <= 0){
      this.setState({time: this.state.time = 0})
    }
  }

  toggleSwitch = (val) => {
    this.setState({switcher: val})
  }

  toggleTheme = (val2) => {
    this.setState({switcher2: val2})
  }

  render() {
    return (
      <View style={this.state.switcher2? styles.containerDarkTheme : styles.containerLightTheme}>
        <StatusBar
        barStyle={this.state.switcher2? 'dark-content' : 'ligth-content'}
        />
        <View style={styles.headers}>
        <Text style={this.state.switcher2? styles.ligthText : styles.darkText}>Cronometer 10 seconds</Text>
        </View>

        <Text style={this.state.switcher2? styles.ligthText : styles.darkText}>Allows message in time end?</Text>
        <Text style={this.state.switcher2? styles.ligthText : styles.darkText}>{this.state.switcher? 'Yes' : 'No'}</Text>

        <View  style={styles.switcher}>
        <Switch
        value={this.state.switcher}
        onValueChange={this.toggleSwitch.bind(this)}
        />
        </View>

        <View style={styles.button}>
        <Button
        title='Start'
        onPress={this.startCount}
        />
        </View>

        <Text style={this.state.switcher2? styles.ligthText : styles.darkText}>{this.state.time <= 0? 'Time ends' : ''}</Text>
        <Text style={this.state.switcher2? styles.ligthText : styles.darkText}>Toogle theme</Text>

        <View  style={styles.switcher}>
        <Switch
        value = {this.state.switcher2}
        onValueChange = {this.toggleTheme.bind(this)}
        />
        </View>

        <View style={styles.container2}>
          <Text style={this.state.switcher2? styles.ligthText : styles.darkText}>Time: {this.state.time}</Text>
        </View>
      </View>
    )
  }
}

const styles = StyleSheet.create({
  containerDarkTheme: {
    flex: 1,
    width: '100%',
    alignSelf: 'center',
    justifyContent: 'flex-start',
    padding: 20,
    backgroundColor: '#222',
    color: '#fff'
  },
  containerLightTheme: {
    flex: 1,
    width: '100%',
    alignSelf: 'center',
    justifyContent: 'flex-start',
    padding: 20,
    backgroundColor: '#fff',
  },
  container2:{
    flex: 1,
    alignSelf: 'center'
  },
  ligthText: {
    padding: 10,
    fontSize: 20,
    color: 'white'
  },
  darkText: {
    padding: 10,
    fontSize: 20,
  },
  switcher: {
    flex: .1,
    flexDirection: 'row',
    justifyContent: 'flex-start'
  },
  headers: {
    flex: .15,
    flexDirection: 'row',
    alignSelf: 'center',
  },
  button: {
    flex: .1,
    flexDirection: 'column',
    justifyContent: 'center',
    paddingTop: 30,
  }
})
