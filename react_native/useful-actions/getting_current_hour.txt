
STATE

import React, { Component } from 'react'
import { Text, View } from 'react-native'

export default class Time extends Component {
  constructor(props){
    super(props);
    this.state = {
      date: new Date()
    }
    setInterval(() => {
      this.setState({date: new Date()})
    }, 1000 );
  }

  render() {
    return (
      <View>
        <Text>Hour: {this.state.date.toLocaleTimeString()}</Text>
      </View>
    )
  }
}
