  function provideArrayRange(start, stop) {
    return Array.from(
      { length: (stop - start) / 1 + 1 },

    );

function provideYearsInterval(start, stop, step) {
  return Array.from(
    { length: (stop - start) / step + 1 },
    (_, i) => start + i * step
  );
}
