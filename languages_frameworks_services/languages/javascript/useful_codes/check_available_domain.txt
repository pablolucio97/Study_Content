import { NextApiRequest, NextApiResponse } from "next";
import puppeteer from "puppeteer";

export default async function (
  req: NextApiRequest,
  res: NextApiResponse
): Promise<void> {
  if (req.method === "POST") {
    try {
      (async () => {
        const { domain } = req.body;
        const browser = await puppeteer.launch();
        const page = await browser.newPage();
        await page.goto(`https://www.whois.com/whois/${domain}`);
        if ((await page.$("#your_id_element")) !== null) {
          res.send("Domínio não disponível");
        } else {
          res.send("Domínio disponível");
        }
      })();
    } catch (error) {
      res.send("Domínio dispoível");
    }
  } else {
    return res.status(403).send("Mehtod not allowed");
  }
}
