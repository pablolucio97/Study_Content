---------CREATING COMPONENTS WITH NATIVE HTML ELEMENT PROPS----------------

1) Create your component. You need to import InputHTMLAttributes and assing 
it with a type, get all props with {...rest} and pass it to the component.

import React, {InputHTMLAttributes} from 'react';

import { Container } from './styles';

type IInput = InputHTMLAttributes<HTMLInputElement>;

const Input: React.FC<IInput> = ({...rest}) => {
  return (
    <Container {...rest}>
    </Container>

  );
}

export default Input;

--------------------------------------------------------------------------

2) Using in the component:

import React, {InputHTMLAttributes} from 'react';

import { InputElement } from './styles';

type IInput = InputHTMLAttributes<HTMLInputElement>;

const Input: React.FC<IInput> = ({...rest}) => {
  return (
    <InputElement {...rest}>
    </InputElement>

  );
}

export default Input;
